    <script>
        const yesBtn = document.getElementById('yesBtn');
        const noBtn = document.getElementById('noBtn');
        const content = document.getElementById('content');
        const success = document.getElementById('success');
        
        let scale = 1;
        let fontSize = 1.2;

        const phrases = [
            "Ты уверена?",
            "А если подумать?",
            "Ну пожалуйста...",
            "Я буду плакать :(",
            "Кнопка 'Да' сама себя не нажмет!",
            "Сдавайся!"
        ];
        let phraseIndex = 0;

        noBtn.addEventListener('click', () => {
            // Увеличиваем масштаб кнопки "Да"
            scale += 0.5;
            fontSize += 0.3;
            yesBtn.style.transform = `scale(${scale})`;
            
            // Меняем текст на кнопке "Нет" для забавы
            noBtn.innerText = phrases[phraseIndex];
            phraseIndex = (phraseIndex + 1) % phrases.length;

            // Если кнопка стала слишком большой, перекрываем всё
            if (scale > 15) {
                yesBtn.style.position = 'fixed';
                yesBtn.style.top = '0';
                yesBtn.style.left = '0';
                yesBtn.style.width = '100vw';
                yesBtn.style.height = '100vh';
                yesBtn.style.zIndex = '100';
            }
        });

        yesBtn.addEventListener('click', () => {
            content.style.display = 'none';
            success.style.display = 'block';
            document.body.style.backgroundColor = '#ffc8dd';
        });
    </script>
